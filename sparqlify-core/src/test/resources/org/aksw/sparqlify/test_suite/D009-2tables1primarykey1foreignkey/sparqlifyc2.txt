Create TriplesMap1 As
  Construct {
    ?s foaf:name ?n
  }
  With
    ?s = uri(exr:student_, spy:urlEncode(?Name))
    ?n = plainLiteral(?Name)
  From
    [[SELECT "Name", COUNT("Sport") FROM "Student" GROUP BY "Name"]]
